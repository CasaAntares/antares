let carrito = [];

function agregarAlCarrito(id, nombre, precio) {
    const productoEnCarrito = carrito.find(item => item.id === id);

    if (productoEnCarrito) {
        productoEnCarrito.cantidad++;
    } else {
        carrito.push({
            id,
            nombre,
            precio,
            cantidad: 1
        });
    }

    actualizarCarrito();
}
function actualizarCarrito() {
            const listaProductos = document.getElementById('productosCarrito');
            listaProductos.innerHTML = '';

            let totalProductos = 0;
            let precioTotal = 0;

    carrito.forEach(producto => {
        const { cantidad, nombre, precio } = producto;
        const productoHTML = `
            <div>
                <button onclick="restarCantidad(${producto.id})">-</button>
                <span class="cantidad">x${cantidad}</span>
                <button onclick="sumarCantidad(${producto.id})">+</button>
                <span class="nombre">│${nombre}│</span>
                <span class="precio">$${precio * cantidad}</span>
            </div>`;
        listaProductos.innerHTML += productoHTML;

        totalProductos += cantidad;
        precioTotal += precio * cantidad;
    });

    document.getElementById('cantidadTotal').textContent = `Total de productos: ${totalProductos}`;
    document.getElementById('precioTotal').textContent = `Precio Total: $${precioTotal}`;
}

            producto.cantidad--;

            carrito = carrito.filter(item => item.id !== id);
        }
        actualizarCarrito();
    }
}

    function habilitarDesplazamientoHorizontal() {
        const elementosProductos = document.querySelectorAll('.productos');

        elementosProductos.forEach(elemento => {
            let isDown = false;
            let startX;
            let scrollLeft;

            elemento.addEventListener('mousedown', e => {
                isDown = true;
                startX = e.pageX - elemento.offsetLeft;
                scrollLeft = elemento.scrollLeft;
            });

            elemento.addEventListener('mouseleave', () => {
                isDown = false;
            });

            elemento.addEventListener('mouseup', () => {
                isDown = false;
            });

            elemento.addEventListener('mousemove', e => {
                if (!isDown) return;
                e.preventDefault();
                const x = e.pageX - elemento.offsetLeft;
                const walk = (x - startX) * 2;
                elemento.scrollLeft = scrollLeft - walk;
            });
        });
    }
